best_clusters=[]
for i in range(1,11):
    kmean=KMeans(n_clusters=i, init='k-means++')
    kmean.fit(df[['Annual Income (k$)','Spending Score (1-100)']])
    kmean.inertia_
    best_clusters.append(kmean.inertia_)
